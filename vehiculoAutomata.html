Proyecto: Vehículo Autómata con Arduino

CITZapopan · Instrucciones del proyecto

🎯 Objetivo

El alumno realizará un vehículo autómata elaborado con Arduino, capaz de detectar obstáculos y tomar decisiones de movimiento automáticamente.

🧰 Materiales

  • 1 Arduino Uno
  • 2 mini protoboards
  • 1 sensor de distancia ultrasónico
  • 2 motores DC (llantas amarillas)
  • 1 tarjeta L293D
  • 1 portapilas
  • 4 pilas AA
  • 25 cables Dupont macho-macho

💻 Código en texto (Arduino)

int DISTANCIA = 0;

long readUltrasonicDistance(int triggerPin, int echoPin)
{
  pinMode(triggerPin, OUTPUT);
  digitalWrite(triggerPin, LOW);
  delayMicroseconds(2);

  digitalWrite(triggerPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(triggerPin, LOW);

  pinMode(echoPin, INPUT);
  return pulseIn(echoPin, HIGH);
}

void setup()
{
  Serial.begin(9600);
  pinMode(5, OUTPUT);
  pinMode(2, OUTPUT);
  pinMode(3, OUTPUT);
  pinMode(4, OUTPUT);
}

void loop()
{
  DISTANCIA = 0.01723 * readUltrasonicDistance(10, 11);
  Serial.println(DISTANCIA);

  if (DISTANCIA < 20) {
    digitalWrite(5, HIGH);
    digitalWrite(2, LOW);
    digitalWrite(3, HIGH);
    digitalWrite(4, LOW);
  } 
  else {
    digitalWrite(5, LOW);
    digitalWrite(2, LOW);
    digitalWrite(3, LOW);
    digitalWrite(4, LOW);

    delay(1000);

    digitalWrite(5, HIGH);
    digitalWrite(2, LOW);
    digitalWrite(3, LOW);
    digitalWrite(4, HIGH);

    delay(400);
  }
}
